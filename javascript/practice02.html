1. 구구단 출력하기

1.1 for문을 이용하여 2단부터 9단까지 출력하시오

{
    for(let i=2; i<=9; i++){
        document.write("<br>")
        for(let j=1; j<=9; j++){
            let sum = i * j;
            document.write(i + " * " + j + " = " + sum, "<br>");
        }
    }
}

1.2 while문을 이용하여 2단부터 9단까지 출력하시오

{
    let i = 2;
    while(i <= 9){
        document.write("<br>")
        let j = 1;
        while(j <= 9){
            sum = i * j
            document.write(i + " * " + j + " = " + sum, "<br>")
            j++;
        }
        i++;
    }
}

1.3 사용자가 원하는 단을 받아 출력하시오

조건 : 사용자가 원하는 단은 prompt()를 사용합니다. 클릭

{
    let num, result;

    //버튼을 클릭하면 경고창
    document.querySelector(".btn03").addEventListener("click", () => {
        num = prompt("출력할 구구단 숫자를 입력해주세요", "2"/*기본값*/);
        //console.log(num)

        for(let i=1; i<=9; i++){
            result = num + " * " + i + " = " + (num * i) + "<br>";
            //console.log(result)
            document.querySelector(".result03").innerHTML += result;
        }
    });
}
